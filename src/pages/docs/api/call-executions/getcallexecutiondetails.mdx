<ApiSection title="Path parameters">
  <ParamField path="call_execution_id" type="string" required>
    The unique identifier (UUID) of the call execution to retrieve. This ID is returned when a test execution creates call executions, and can also be found in the call executions list within a test execution's details.
  </ParamField>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="id" type="string">
    The unique identifier (UUID) of the call execution.
  </ResponseField>
  <ResponseField name="service_provider_call_id" type="string or null">
    The external call identifier assigned by the telephony or service provider (e.g., Twilio, Vonage). This is `null` for text agent calls that do not go through an external provider.
  </ResponseField>
  <ResponseField name="session_id" type="string or null">
    The session identifier linking this call execution to a broader conversation session. Useful for tracking multi-turn or multi-call interactions. `null` if no session context applies.
  </ResponseField>
  <ResponseField name="status" type="string">
    The current lifecycle status of the call execution. Possible values are `pending` (waiting to start), `queued` (scheduled for execution), `ongoing` (currently in progress), `completed` (finished successfully), `failed` (encountered an error), `analyzing` (post-call evaluation in progress), or `cancelled` (manually stopped before completion).
  </ResponseField>
  <ResponseField name="duration" type="number or null">
    The total duration of the call in seconds, measured from the moment the connection was established to when it ended. `null` if the call has not yet completed or failed before connecting.
  </ResponseField>
  <ResponseField name="start_time" type="datetime or null">
    The ISO 8601 timestamp of when the call was actually connected and the conversation began. `null` if the call has not yet started or failed before connecting.
  </ResponseField>
  <ResponseField name="timestamp" type="datetime">
    The ISO 8601 timestamp of when the call execution record was created in the system.
  </ResponseField>
  <ResponseField name="call_type" type="string or null">
    The direction of the call. Either `Inbound` (the agent received the call) or `Outbound` (the agent initiated the call). `null` for text agents where call direction does not apply.
  </ResponseField>
  <ResponseField name="transcript" type="array">
    An ordered array of conversation turns between the agent and the simulated user. For voice agents, this is the speech-to-text transcription of the audio.
  </ResponseField>
  <ApiCollapsible title="Show 2 properties">
    <ResponseField name="role" type="string">
      The speaker for this turn. Possible values: `user` (the simulated customer), `assistant` (the agent being tested), `system` (system-level messages), `tool_calls` (tool invocation requests), or `tool_call_result` (tool execution results).
    </ResponseField>
    <ResponseField name="content" type="string">
      The text content of the utterance or message for this turn.
    </ResponseField>
  </ApiCollapsible>
  <ResponseField name="scenario" type="string">
    The human-readable name of the scenario that was used for this call execution. This corresponds to the scenario configuration that defined the test case parameters and expected behavior.
  </ResponseField>
  <ResponseField name="scenario_id" type="string or null">
    The unique identifier (UUID) of the scenario associated with this call execution. `null` in rare cases where the scenario reference has been removed.
  </ResponseField>
  <ResponseField name="scenario_columns" type="array">
    An array of objects representing the dataset column values that were used to parameterize this specific call. Each object contains the column name and its value for the test case row that drove this execution.
  </ResponseField>
  <ResponseField name="overall_score" type="number or null">
    The aggregated overall evaluation score for this call, computed from all configured evaluation metrics. Typically on a 0-10 scale. `null` if evaluation has not been run or no evaluations are configured.
  </ResponseField>
  <ResponseField name="response_time" type="number or null">
    The average response time of the agent in seconds, measuring how quickly the agent replied after each user turn. `null` if not measured or not applicable.
  </ResponseField>
  <ResponseField name="eval_outputs" type="object or null">
    A structured object containing the detailed results of all evaluations that were run on this call. Keys are evaluation names (e.g., `tone_check`, `accuracy`) and values are objects with `score`, `explanation`, and other evaluation-specific fields. `null` if no evaluations have been executed.
  </ResponseField>
  <ResponseField name="eval_metrics" type="object or null">
    Additional evaluation metric data that supplements the primary eval outputs. Contains computed aggregations and derived metrics from the evaluation pipeline. `null` if unavailable.
  </ResponseField>
  <ResponseField name="audio_url" type="string or null">
    A direct URL to the audio recording of the call. Only available for voice agent calls that have completed recording and processing. `null` for text agents or if the recording is not yet available.
  </ResponseField>
  <ResponseField name="recordings" type="object or null">
    Provider-specific recording URLs and metadata. May contain multiple URLs for different recording formats or channels depending on the telephony provider used. `null` if no recordings are available.
  </ResponseField>
  <ResponseField name="customer_name" type="string or null">
    The name of the simulated customer persona used during the call. `null` if no customer persona was assigned.
  </ResponseField>
  <ResponseField name="call_summary" type="string or null">
    An AI-generated or manually provided summary of the call conversation, capturing the key topics discussed and outcomes. `null` if summary generation has not been performed.
  </ResponseField>
  <ResponseField name="ended_reason" type="string or null">
    The reason the call ended (e.g., `customer_hangup`, `agent_hangup`, `timeout`, `error`). Useful for understanding whether the call concluded naturally or was interrupted. `null` if the call has not ended or the reason was not captured.
  </ResponseField>
  <ResponseField name="simulatorAgentName" type="string or null">
    The name of the simulator agent that played the role of the customer during this call. `null` if no dedicated simulator agent was used.
  </ResponseField>
  <ResponseField name="simulatorAgentId" type="string or null">
    The unique identifier (UUID) of the simulator agent used for this call execution. `null` if no dedicated simulator agent was assigned.
  </ResponseField>
  <ResponseField name="agent_definition_used_name" type="string or null">
    The name of the agent definition (the agent under test) that was evaluated during this call. `null` if the agent definition reference is unavailable.
  </ResponseField>
  <ResponseField name="agent_definition_used_id" type="string or null">
    The unique identifier (UUID) of the agent definition used in this call execution. `null` if the agent definition reference is unavailable.
  </ResponseField>
  <ResponseField name="tool_outputs" type="object or null">
    The outputs from any tool calls made by the agent during the conversation. Contains the tool invocation details and their results. `null` if no tool calls were made or tool evaluation was not enabled.
  </ResponseField>
  <ResponseField name="rerun_snapshots" type="array">
    An array of historical snapshots from previous reruns of this call execution. Each snapshot captures the state of the call (transcript, metrics, evaluation results) at the time the rerun was initiated, allowing you to compare performance across iterations.
  </ResponseField>
  <ResponseField name="avg_agent_latency" type="number or null">
    The average latency of the agent's responses in seconds, measured across all turns in the conversation. Lower values indicate faster response times. Only applicable to voice agents. `null` if not measured.
  </ResponseField>
  <ResponseField name="user_interruption_count" type="integer or null">
    The total number of times the simulated user interrupted the agent during the call. High interruption counts may indicate the agent is speaking too slowly or providing irrelevant information. Voice agents only.
  </ResponseField>
  <ResponseField name="user_interruption_rate" type="number or null">
    The proportion of agent turns that were interrupted by the user, expressed as a decimal between 0 and 1. For example, 0.15 means 15% of agent turns were interrupted. Voice agents only.
  </ResponseField>
  <ResponseField name="user_wpm" type="number or null">
    The simulated user's average speaking rate in words per minute. Useful for understanding the pace of the customer side of the conversation. Voice agents only.
  </ResponseField>
  <ResponseField name="bot_wpm" type="number or null">
    The agent's average speaking rate in words per minute. Useful for tuning the agent's speech speed for natural conversation flow. Voice agents only.
  </ResponseField>
  <ResponseField name="talk_ratio" type="number or null">
    The proportion of total talk time attributed to the agent, expressed as a decimal between 0 and 1. A value of 0.6 means the agent spoke for 60% of the total conversation time. Voice agents only.
  </ResponseField>
  <ResponseField name="ai_interruption_count" type="integer or null">
    The total number of times the agent interrupted the simulated user during the call. High values may indicate the agent is not waiting for the user to finish speaking. Voice agents only.
  </ResponseField>
  <ResponseField name="ai_interruption_rate" type="number or null">
    The proportion of user turns that were interrupted by the agent, expressed as a decimal between 0 and 1. Voice agents only.
  </ResponseField>
  <ResponseField name="avg_stop_time_after_interruption" type="number or null">
    The average time in seconds the agent takes to stop speaking after being interrupted by the user. Lower values indicate the agent is more responsive to interruptions. Voice agents only.
  </ResponseField>
  <ResponseField name="total_tokens" type="integer or null">
    The total number of tokens consumed during the conversation, combining both input and output tokens. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="input_tokens" type="integer or null">
    The number of input (prompt) tokens sent to the language model across all turns. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="output_tokens" type="integer or null">
    The number of output (completion) tokens generated by the language model across all turns. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="avg_latency_ms" type="number or null">
    The average response latency in milliseconds for chat/text agent turns. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="turn_count" type="integer or null">
    The total number of conversation turns (user-agent exchanges) in the chat session. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="csat_score" type="number or null">
    The customer satisfaction score computed for the conversation, typically on a 1-5 or 1-10 scale. Available within `conversation_metrics_data` for chat/text agents only.
  </ResponseField>
  <ResponseField name="stt_cost" type="number or null">
    The cost in USD incurred for speech-to-text transcription during the call. Only applicable to voice agents. `null` if not tracked.
  </ResponseField>
  <ResponseField name="llm_cost" type="number or null">
    The cost in USD incurred for language model inference during the call. Includes all token-based charges for the conversation.
  </ResponseField>
  <ResponseField name="tts_cost" type="number or null">
    The cost in USD incurred for text-to-speech synthesis during the call. Only applicable to voice agents. `null` if not tracked.
  </ResponseField>
  <ResponseField name="storage_cost" type="number or null">
    The cost in USD incurred for storing call recordings and related artifacts. `null` if not tracked.
  </ResponseField>
  <ResponseField name="total_cost" type="number or null">
    The total combined cost in USD for the entire call execution, summing STT, LLM, TTS, and storage costs. `null` if cost tracking is not enabled.
  </ResponseField>
</ApiSection>
