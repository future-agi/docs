<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="id" type="string">
    The unique identifier (UUID) of the scenario.
  </ResponseField>
  <ResponseField name="name" type="string">
    The human-readable name of the scenario.
  </ResponseField>
  <ResponseField name="description" type="string">
    The description of the scenario, if one was provided during creation.
  </ResponseField>
  <ResponseField name="source" type="string">
    The data source used to create the scenario (e.g., the dataset name or script reference).
  </ResponseField>
  <ResponseField name="scenario_type" type="string">
    The type of scenario: `dataset`, `script`, or `graph`.
  </ResponseField>
  <ResponseField name="dataset_id" type="string">
    The UUID of the underlying dataset that holds the scenario's test case rows. `null` if the scenario does not have an associated dataset.
  </ResponseField>
  <ResponseField name="agent_type" type="string">
    The type of agent this scenario is designed for (`voice` or `text`).
  </ResponseField>
  <ResponseField name="status" type="string">
    The current status of the scenario. Common values include `processing` (still being generated), `completed` (ready for use), or `failed` (generation encountered an error).
  </ResponseField>
  <ResponseField name="graph" type="object">
    The conversation graph structure associated with the scenario. This defines the conversation flow, decision nodes, and expected paths. Empty object `{}` if no graph is configured.
  </ResponseField>
  <ResponseField name="prompts" type="array">
    An array of prompt objects used by the simulator agent to instruct the simulated customer on how to behave during test calls.
  </ResponseField>
  <ApiCollapsible title="Show 2 properties">
    <ResponseField name="role" type="string">
      The role of the prompt message (e.g., `"system"`).
    </ResponseField>
    <ResponseField name="content" type="string">
      The prompt text that instructs the simulated customer's behavior.
    </ResponseField>
  </ApiCollapsible>
  <ResponseField name="dataset_rows" type="integer">
    The number of test case rows in the scenario's dataset. This corresponds to the number of individual test cases available for execution.
  </ResponseField>
  <ResponseField name="created_at" type="datetime">
    The ISO 8601 timestamp of when the scenario was created.
  </ResponseField>
  <ResponseField name="updated_at" type="datetime">
    The ISO 8601 timestamp of the most recent modification to the scenario.
  </ResponseField>
</ApiSection>
