<ApiSection title="Request body">
  <ParamField body="name" type="string" optional>
    An updated name for the scenario. Maximum 255 characters. Cannot be blank if provided.
  </ParamField>

  <ParamField body="description" type="string" optional>
    An updated description for the scenario.
  </ParamField>

  <ParamField body="graph" type="object" optional>

  </ParamField>

  <ParamField body="prompt" type="string" optional>
    An updated system prompt for the simulator agent. This prompt instructs the simulated customer on how to behave during test calls. Supports template variables `{{persona}}` (replaced with the assigned persona description) and `{{situation}}` (replaced with the test case context).
  </ParamField>
</ApiSection>
