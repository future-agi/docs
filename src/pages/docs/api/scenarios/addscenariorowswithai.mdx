<ApiSection title="Path parameters">
  <ParamField path="scenario_id" type="string" required>
    The unique identifier (UUID) of the scenario to add rows to. The scenario must have an associated dataset.
  </ParamField>
</ApiSection>

<ApiSection title="Request body">
  <ParamField body="numRows" type="integer" required>
    The number of new rows to generate and add to the scenario's dataset. Accepts values between `10` and `100`. Each row represents an individual test case with AI-generated data populated across all existing columns in the dataset.
  </ParamField>

  <ParamField body="description" type="string" optional>
    An optional description that guides the AI when generating content for the new rows. Be specific about the types of test cases you want — for example, `"Generate rows covering edge cases like expired subscriptions, disputed charges, and multi-language support requests"`. When omitted, the AI uses the existing rows and column definitions as context.
  </ParamField>
</ApiSection>

  <ResponseField name="dataset_id" type="string">

  </ResponseField>
  <ResponseField name="numRows" type="integer">
