---
title: "Delete Agent Definitions"
description: "Bulk soft-deletes one or more agent definitions and all their associated versions. The records are marked as deleted but retained in the database for auditing purposes."
---

<ApiSection title="Request body">
  <ParamField body="agent_ids" type="array of string" required>
    A list of one or more agent definition UUIDs to mark as deleted. All associated agent versions are also soft-deleted. The list must contain at least one ID. Deleted agents and their versions are excluded from future listings but retained in the database.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">

  </ResponseField>
  <ResponseField name="agents_updated" type="integer">
    The number of agent definitions that were marked as deleted.
  </ResponseField>
  <ResponseField name="versions_updated" type="integer">
    The total number of agent versions that were marked as deleted across all specified agent definitions.
  </ResponseField>
</ApiSection>
