<ApiSection title="Path parameters">
  <ParamField path="test_execution_id" type="string" required>
    The unique identifier (UUID) of the test execution for which to retrieve the evaluation explanation summary. The test execution should ideally be in a `completed` or `evaluating` state. If the summary has not been generated yet, this request automatically triggers asynchronous generation.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="object">
    The wrapper object containing the evaluation explanation summary data and its generation metadata.
  </ResponseField>
  <ApiCollapsible title="Show 3 properties">
    <ResponseField name="response" type="object or null">
      The evaluation explanation summary content, produced by analyzing all call execution evaluation results within the test execution. Contains a high-level summary of agent performance, critical issues identified across calls, and actionable recommendations. Returns `null` if the summary has not yet been generated — in this case, check the `status` field and poll again after a short delay.
    </ResponseField>
    <ResponseField name="last_updated" type="datetime or null">
      The ISO 8601 timestamp of when the evaluation explanation summary was last generated or refreshed. `null` if the summary has never been generated. Use this to determine if you need to trigger a regeneration after new evaluations have been run.
    </ResponseField>
    <ResponseField name="status" type="string">
      The current status of the summary generation process. Possible values are `pending` (queued for generation), `running` (actively being generated by the AI analysis pipeline), `completed` (summary is ready and available in the `response` field), or `failed` (generation encountered an error and could not complete).
    </ResponseField>
  </ApiCollapsible>
  <ResponseField name="success" type="boolean">
    Indicates whether the API request was processed successfully. Returns `true` even when the summary is still being generated -- check `data.status` to determine if the summary content is available.
  </ResponseField>
</ApiSection>
