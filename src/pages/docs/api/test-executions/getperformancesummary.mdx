<ApiSection title="Authentication">
  <ParamField name="X-Api-Key" type="string" required>
    Your Future AGI API key used to authenticate requests. You can find and manage your API keys in the [Dashboard](https://app.futureagi.com) under Settings.
  </ParamField>
  <ParamField name="X-Secret-Key" type="string" required>
    Your Future AGI secret key, used alongside the API key for request authentication. This is generated when you create an API key in the [Dashboard](https://app.futureagi.com).
  </ParamField>
</ApiSection>

<ApiSection title="Path parameters">
  <ParamField path="test_execution_id" type="string" required>
    The unique identifier (UUID) of the test execution for which to retrieve the performance summary. The summary is most meaningful when the test execution is in a `completed` state, though partial results are available for in-progress executions. You can obtain this ID from the test execution list or details endpoints.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="test_run_performance_metrics" type="object">
    An aggregated view of the overall test run performance, including pass and fail rates computed across all call executions within this test execution.
  </ResponseField>
  <ResponseField name="test_run_performance_metrics.pass_rate" type="number">
    The percentage of call executions that completed successfully (i.e., reached `completed` status), expressed as a value between 0 and 100. For example, a value of 88.0 means 88% of all calls in the test execution finished without errors.
  </ResponseField>
  <ResponseField name="test_run_performance_metrics.total_test_runs" type="integer">
    The total number of call executions within this test execution, including both completed and failed calls. This represents the total test case count for the run.
  </ResponseField>
  <ResponseField name="test_run_performance_metrics.latest_fail_rate" type="number">
    The percentage of call executions that ended in a `failed` status, expressed as a value between 0 and 100. This is the complement of `pass_rate` when all calls have reached a terminal state. A high fail rate may indicate agent configuration issues, provider connectivity problems, or scenario design errors.
  </ResponseField>
  <ResponseField name="top_performing_scenarios" type="array">
    An array of the top-performing scenarios ranked by their average performance score in descending order. Returns up to 4 scenarios, allowing you to quickly identify which test scenarios the agent handles best. Each object in the array contains the scenario details and its aggregated performance metrics.
  </ResponseField>
  <ResponseField name="top_performing_scenarios[].scenario_name" type="string">
    The human-readable name of the scenario. This matches the name defined in the scenario configuration.
  </ResponseField>
  <ResponseField name="top_performing_scenarios[].test_count" type="integer">
    The number of call executions that were run for this specific scenario. This reflects how many test case rows from the scenario dataset were executed.
  </ResponseField>
  <ResponseField name="top_performing_scenarios[].performance_score" type="number">
    The average overall evaluation score for all call executions within this scenario, on a 0-10 scale. This is computed as the mean of individual call `overall_score` values. Higher scores indicate better agent performance on the scenario's test cases.
  </ResponseField>
</ApiSection>

<ApiSection title="Errors">
  <ParamField name="401" type="Unauthorized">
    The request could not be authenticated. Verify that both `X-Api-Key` and `X-Secret-Key` headers are present and contain valid, non-expired credentials. Ensure the API key has access to the workspace that owns this test execution.
  </ParamField>
  <ParamField name="404" type="Not Found">
    No test execution was found with the specified ID. The test execution may have been deleted, or the UUID may be incorrect. Verify the `test_execution_id` value and ensure it belongs to your organization.
  </ParamField>
  <ParamField name="500" type="Internal Server Error">
    An unexpected server-side error occurred while computing the performance summary. If this persists, contact support with the request details and the `test_execution_id`.
  </ParamField>
</ApiSection>
