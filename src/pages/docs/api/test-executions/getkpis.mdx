<ApiSection title="Authentication">
  <ParamField name="X-Api-Key" type="string" required>
    Your Future AGI API key used to authenticate requests. You can find and manage your API keys in the [Dashboard](https://app.futureagi.com) under Settings.
  </ParamField>
  <ParamField name="X-Secret-Key" type="string" required>
    Your Future AGI secret key, used alongside the API key for request authentication. This is generated when you create an API key in the [Dashboard](https://app.futureagi.com).
  </ParamField>
</ApiSection>

<ApiSection title="Path parameters">
  <ParamField path="test_execution_id" type="string" required>
    The unique identifier (UUID) of the test execution for which to retrieve aggregated KPI metrics. The response structure varies depending on whether the test execution is for a voice agent or a text agent. You can obtain this ID from the test execution list or details endpoints.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="total_calls" type="integer">
    The total number of call executions created within this test execution, including both successful and failed calls.
  </ResponseField>
  <ResponseField name="avg_score" type="number">
    The average overall evaluation score across all completed call executions in this test execution. Computed as the mean of individual call `overall_score` values.
  </ResponseField>
  <ResponseField name="avg_response" type="number">
    The average response time in seconds across all completed call executions. For voice agents, this measures the time between the user finishing speaking and the agent beginning to respond. For text agents, this measures API response latency.
  </ResponseField>
  <ResponseField name="callsAttempted" type="integer">
    The total number of calls that were attempted (initiated) during this test execution, regardless of whether they connected successfully.
  </ResponseField>
  <ResponseField name="connectedCalls" type="integer">
    The number of calls that successfully connected and produced a conversation. This excludes calls that failed before a connection was established (e.g., timeout, provider errors).
  </ResponseField>
  <ResponseField name="callsConnectedPercentage" type="number">
    The percentage of attempted calls that successfully connected, calculated as `(connectedCalls / callsAttempted) * 100`. A low percentage may indicate infrastructure or provider issues.
  </ResponseField>
  <ResponseField name="failed_calls" type="integer">
    The number of call executions that ended in a `failed` status due to errors during connection, conversation, or evaluation.
  </ResponseField>
  <ResponseField name="total_duration" type="integer">
    The total combined duration in seconds of all call executions in this test execution. Useful for understanding the overall time investment of the test run.
  </ResponseField>
  <ResponseField name="agent_type" type="string">
    The type of agent being tested. Either `voice` (telephony-based voice agent) or `text` (text-based agent). This determines which additional metric fields are present in the response.
  </ResponseField>
  <ResponseField name="is_inbound" type="boolean or null">
    Indicates whether the test execution used inbound calls (`true`) or outbound calls (`false`). `null` for text agents where call direction does not apply.
  </ResponseField>
  <ResponseField name="scenarioGraphs" type="object">
    A structured object containing per-scenario performance data suitable for visualization. Keys are scenario identifiers and values contain aggregated metrics for that scenario, enabling comparison of agent performance across different test scenarios.
  </ResponseField>
  <ResponseField name="avg_agent_latency" type="number">
    The average agent response latency in seconds across all voice calls. Measures the time from when the user stops speaking to when the agent begins responding. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_user_interruption_count" type="number">
    The average number of times the simulated user interrupted the agent per call. High values may suggest the agent is providing slow or irrelevant responses. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_user_interruption_rate" type="number">
    The average proportion of agent turns interrupted by the user across all calls, expressed as a decimal between 0 and 1. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_user_wpm" type="number">
    The average words-per-minute speaking rate of the simulated user across all voice calls. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_bot_wpm" type="number">
    The average words-per-minute speaking rate of the agent across all voice calls. Useful for tuning the agent's speech pacing. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_talk_ratio" type="number">
    The average proportion of total talk time attributed to the agent across all calls, expressed as a decimal between 0 and 1. A balanced conversation typically has a ratio around 0.4-0.6. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_ai_interruption_count" type="number">
    The average number of times the agent interrupted the simulated user per call. High values may indicate the agent is not properly detecting end-of-turn signals. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_ai_interruption_rate" type="number">
    The average proportion of user turns interrupted by the agent across all calls, expressed as a decimal between 0 and 1. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_stop_time_after_interruption" type="number">
    The average time in seconds the agent takes to stop speaking after being interrupted by the user, averaged across all calls. Lower values indicate better responsiveness to user interruptions. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="agentTalkPercentage" type="number">
    The percentage of total conversation time the agent was speaking, averaged across all calls. This is the `avg_talk_ratio` expressed as a percentage (0-100). Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="customerTalkPercentage" type="number">
    The percentage of total conversation time the simulated user was speaking, averaged across all calls. This is the complement of `agentTalkPercentage`. Only present for voice agent test executions.
  </ResponseField>
  <ResponseField name="avg_total_tokens" type="number">
    The average total tokens (input + output) consumed per call execution. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_input_tokens" type="number">
    The average number of input (prompt) tokens sent to the language model per call. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_output_tokens" type="number">
    The average number of output (completion) tokens generated by the language model per call. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_chat_latency_ms" type="number">
    The average response latency in milliseconds for text agent turns across all calls. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_turn_count" type="number">
    The average number of conversation turns (user-agent exchanges) per call. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_csat_score" type="number">
    The average customer satisfaction score across all calls. Only present for text agent test executions.
  </ResponseField>
  <ResponseField name="avg_[metric_name]" type="number">
    Dynamically generated average fields for each evaluation metric configured on the test run. For example, if you have evaluations named `tone_check` and `accuracy`, the response will include `avg_tone_check` and `avg_accuracy` fields with the mean scores across all call executions.
  </ResponseField>
</ApiSection>

<ApiSection title="Errors">
  <ParamField name="401" type="Unauthorized">
    The request could not be authenticated. Verify that both `X-Api-Key` and `X-Secret-Key` headers are present and contain valid, non-expired credentials. Ensure the API key has access to the workspace that owns this test execution.
  </ParamField>
  <ParamField name="404" type="Not Found">
    No test execution was found with the specified ID. The test execution may have been deleted, or the UUID may be incorrect. Verify the `test_execution_id` value and ensure it belongs to your organization.
  </ParamField>
  <ParamField name="500" type="Internal Server Error">
    An unexpected server-side error occurred while computing the KPI metrics. If this persists, contact support with the request details and the `test_execution_id`.
  </ParamField>
</ApiSection>
