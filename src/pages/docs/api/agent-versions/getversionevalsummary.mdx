<ApiSection title="Response" status={200} statusText="OK">
  Returns an array of evaluation template statistics. Each object in the array represents one configured eval template and its aggregated results across all call executions for this version. Returns an empty array `[]` if no evaluation configurations exist.

  <ResponseField name="eval_id" type="string">
    The unique identifier (UUID) of the evaluation template.
  </ResponseField>
  <ResponseField name="eval_name" type="string">
    The human-readable name of the evaluation template (e.g., `"tone-check"`, `"relevance"`).
  </ResponseField>
  <ResponseField name="eval_type" type="string">
    The type category of the evaluation (e.g., `"tone"`, `"relevance"`, `"accuracy"`).
  </ResponseField>
  <ResponseField name="total" type="integer">
    The total number of evaluations run for this template across all call executions.
  </ResponseField>
  <ResponseField name="pass" type="integer">
    The number of evaluations that passed for this template.
  </ResponseField>
  <ResponseField name="fail" type="integer">
    The number of evaluations that failed for this template.
  </ResponseField>
  <ResponseField name="error" type="integer">
    The number of evaluations that encountered errors during execution (distinct from pass/fail — these are evaluations that could not complete).
  </ResponseField>
  <ResponseField name="pass_rate" type="number">
    The pass rate as a percentage (0-100), calculated as `pass / total * 100`. Use this to quickly compare evaluation quality across templates and versions.
  </ResponseField>
</ApiSection>
