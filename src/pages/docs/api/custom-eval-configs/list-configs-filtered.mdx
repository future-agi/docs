<ApiSection title="Query parameters">
  <ParamField query="project_id" type="string" optional>

  <ParamField query="task_id" type="string" optional>

  </ParamField>

<ApiSection title="Response" status={200} statusText="OK">
  Returns an array of custom eval config objects. Each object contains the following fields:

  <ResponseField name="id" type="string">
    The unique identifier (UUID) of the custom eval config.
  </ResponseField>
  <ResponseField name="eval_template" type="string">
    The UUID of the eval template this config is based on. The template defines the evaluation logic and scoring criteria.
  </ResponseField>
  <ResponseField name="name" type="string">
    The human-readable name assigned to this config when it was created.
  </ResponseField>
  <ResponseField name="config" type="object">
    The configuration dictionary that customizes the eval template's behavior for this specific config. Values are normalized against the template's config schema.
  </ResponseField>
  <ResponseField name="mapping" type="object">
    The field mapping that defines how trace span fields correspond to the eval template's expected inputs (e.g., `input`, `output`, `context`).
  </ResponseField>
  <ResponseField name="project" type="string">
    The UUID of the project this config is associated with.
  </ResponseField>
  <ResponseField name="filters" type="object">
    Any filter criteria applied to this config, which further restrict which spans it evaluates.
  </ResponseField>
  <ResponseField name="error_localizer" type="boolean">
    Whether error localization is enabled for this config. When `true`, the evaluator identifies the specific part of the output that caused a failure.
  </ResponseField>
  <ResponseField name="kb_id" type="string">
    The UUID of the knowledge base file referenced by this config, if any. Used for evaluations that compare output against reference documents.
  </ResponseField>
  <ResponseField name="model" type="string">
    The evaluation model used for scoring, if explicitly set. When `null`, the system default model is used.
  </ResponseField>
  <ResponseField name="eval_group" type="string">
    The name of the eval group this config belongs to, if it has been added to one. Eval groups allow you to organize and apply multiple configs together.
  </ResponseField>
</ApiSection>
