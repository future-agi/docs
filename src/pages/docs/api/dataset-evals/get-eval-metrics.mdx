<ApiSection title="Request body">
  <ParamField body="eval_template_id" type="string" required>
    The universally unique identifier (UUID) of the eval template whose aggregated metrics you want to retrieve. This endpoint also accepts this parameter as a query parameter for GET requests.
  </ParamField>

  <ParamField body="filters" type="array" optional>
    An optional array of filter objects used to scope which evaluation results are included in the metrics calculation. Only results with `SUCCESS` status are considered. When empty or not provided, all successful results for the template are included. For GET requests, pass this as a JSON-encoded string in the `filters` query parameter.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="object">
    The aggregated evaluation metrics for the specified eval template. The exact structure depends on the eval template's output type and configuration, but commonly includes pass/fail rates, score distributions, or choice distributions.

  </ResponseField>
    <ApiCollapsible title="Show 3 properties">
      <ResponseField name="total_evaluations" type="integer">
        The total number of evaluation results included in the metrics calculation, after applying any specified filters. Only results with `SUCCESS` status are counted.
      </ResponseField>

      <ResponseField name="pass_rate" type="number">
        The proportion of evaluations that resulted in a pass outcome, expressed as a decimal between `0` and `1`. For example, `0.85` indicates that 85% of evaluated rows passed. Only present for `Pass/Fail` output type templates.
      </ResponseField>

      <ResponseField name="fail_rate" type="number">
        The proportion of evaluations that resulted in a fail outcome, expressed as a decimal between `0` and `1`. This value is the complement of `pass_rate`. Only present for `Pass/Fail` output type templates.
      </ResponseField>
    </ApiCollapsible>

  <ResponseField name="success" type="boolean">
    A boolean flag indicating whether the request completed successfully. Returns `true` when metrics are calculated without errors.
  </ResponseField>
</ApiSection>
