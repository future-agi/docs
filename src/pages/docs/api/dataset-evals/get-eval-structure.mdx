<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset. Required when `eval_type` is `user`, as the user eval is scoped to a specific dataset. For `preset` and `previously_configured` types, this parameter is still required in the URL path.
  </ParamField>

  <ParamField path="eval_id" type="string" required>
    The universally unique identifier (UUID) of the evaluation template or user eval metric whose structure you want to retrieve. The interpretation of this ID depends on the `eval_type` query parameter.
  </ParamField>
</ApiSection>

<ApiSection title="Query parameters">
  <ParamField query="eval_type" type="string" required>
    The type of evaluation to retrieve the structure for. Accepted values are `preset` (a built-in eval template provided by Future AGI), `user` (a user-created eval configured on a specific dataset), and `previously_configured` (an eval template that has been previously configured on a dataset). This determines how the `eval_id` path parameter is interpreted and which data is returned.
  </ParamField>
</ApiSection>

            The universally unique identifier (UUID) of the underlying eval template that defines the evaluation criteria and output type.
          </ResponseField>

          <ResponseField name="name" type="string">
            The human-readable display name of the evaluation, shown in the dataset's eval sidebar and template selection interfaces.
          </ResponseField>

          <ResponseField name="description" type="string">
            A detailed description of what the evaluation assesses and how it works, helping users understand its purpose and applicability.
          </ResponseField>

          <ResponseField name="required_keys" type="array">
            An array of variable key names that must be mapped to dataset columns when configuring this evaluation. Every key in this list must have a corresponding entry in the `mapping` object.
          </ResponseField>

          <ResponseField name="optional_keys" type="array">
            An array of variable key names that may optionally be mapped to dataset columns. These provide additional context to the evaluation but are not required for execution.
          </ResponseField>

          <ResponseField name="variable_keys" type="array">
            The complete list of all template variable keys (both required and optional) recognized by this evaluation template.
          </ResponseField>

          <ResponseField name="mapping" type="object">
            The current key-to-column mapping configuration. For `user` type evals, this reflects the configured mapping; for `preset` type, this is typically an empty object since no mapping has been configured yet.
          </ResponseField>

          <ResponseField name="config" type="object">
            The eval configuration parameters specific to this template, including any template-specific settings that customize evaluation behavior.
          </ResponseField>

          <ResponseField name="params" type="object">
            The runtime parameters for the evaluation, including thresholds, weights, or other execution-time settings.
          </ResponseField>

          <ResponseField name="output" type="string">
            The output type of the evaluation. One of `Pass/Fail` (binary outcome), `score` (numeric score), or `choices` (selection from predefined options).
          </ResponseField>

          <ResponseField name="choices" type="object">
            The choice options for choice-type evaluations. Contains a key-value mapping of choice identifiers to labels (e.g., `{"A": "Good", "B": "Bad"}`). Returns `null` for non-choice output types.
          </ResponseField>

          <ResponseField name="models" type="string">
            The available or configured model for running this evaluation (e.g., `turing_small`, `turing_large`).
          </ResponseField>

          <ResponseField name="kb_id" type="string">
            The universally unique identifier (UUID) of the associated knowledge base, if one is configured. Returns `null` when no knowledge base is associated.
          </ResponseField>

          <ResponseField name="error_localizer" type="boolean">
            Whether error localization is enabled for this evaluation. When `true`, the evaluation engine identifies specific portions of output that caused failures.
          </ResponseField>

          <ResponseField name="api_key_available" type="boolean">
            Whether the required API key for the evaluation model is configured in your organization's settings. When `false`, the evaluation cannot be run until the appropriate API key is added.
          </ResponseField>
