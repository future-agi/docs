<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>

<ApiSection title="Request body">
  <ParamField body="delete_column" type="boolean" optional>
    Whether to permanently delete the eval's associated column and all its cell data. Default is `false`. When set to `true`, the eval column, reason column, and all associated cell data are soft-deleted, and the column is removed from the dataset's column order. This action cannot be undone. When set to `false`, the eval is hidden from the sidebar without deleting any underlying data, allowing it to be restored later.
  </ParamField>
</ApiSection>
