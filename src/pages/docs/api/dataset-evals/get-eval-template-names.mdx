<ApiSection title="Request body">
  <ParamField body="search_text" type="string" optional>
    An optional text string used to filter eval template names. The search is case-insensitive and matches against template names. When empty or not provided, all eval templates accessible to your organization are returned. For example, searching `"hallucination"` returns all templates whose names contain that word.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="array">
    An array of eval template objects matching the search criteria. Each entry provides the essential identification and description information needed to select a template for use in dataset evaluations.

  </ResponseField>
    <ApiCollapsible title="Show 3 properties">
      <ResponseField name="id" type="string">
        The universally unique identifier (UUID) of the eval template. Use this ID as the `template_id` when adding an evaluation to a dataset.
      </ResponseField>

      <ResponseField name="name" type="string">
        The human-readable name of the eval template, used to identify it in the template selection interface and throughout the platform.
      </ResponseField>

      <ResponseField name="description" type="string">
        A description of what the eval template assesses and how it works, helping users understand the purpose and applicability of the evaluation.
      </ResponseField>
    </ApiCollapsible>

  <ResponseField name="success" type="boolean">
    A boolean flag indicating whether the request completed successfully. Returns `true` when the template names are retrieved without errors.
