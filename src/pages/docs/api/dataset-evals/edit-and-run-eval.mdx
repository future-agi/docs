<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset containing the evaluation to edit. The dataset must exist within your organization and be accessible with your current API credentials.
  </ParamField>

  <ParamField path="eval_id" type="string" required>
    The universally unique identifier (UUID) of the user eval metric to update. The evaluation must exist within the specified dataset, belong to your organization, and not have been deleted.
  </ParamField>
</ApiSection>

<ApiSection title="Request body">
  <ParamField body="config" type="object" required>
    The updated configuration object for the evaluation. This controls how the evaluation is executed against dataset rows, including variable mapping and evaluation parameters.

    <ApiCollapsible title="Show 4 properties">
      <ResponseField name="config" type="object">
        Template-specific configuration parameters that customize the evaluation behavior. The available options depend on the eval template associated with this evaluation.
      </ResponseField>

      <ResponseField name="params" type="object">
        Runtime parameters passed to the evaluation engine during processing. These may include thresholds, weights, or other execution-time settings.
      </ResponseField>

      <ResponseField name="mapping" type="object">
        A key-value mapping that connects the eval template's variable keys to actual dataset column names. All required keys defined by the template must be present in this mapping. For example, `{"input": "question_column", "output": "response_column"}`.
      </ResponseField>

      <ResponseField name="reason_column" type="boolean">
        Whether to create or keep an additional reason column alongside the eval result column. When `true`, the evaluation engine generates an explanation for each evaluation outcome.
      </ResponseField>
    </ApiCollapsible>
  </ParamField>

  <ParamField body="kb_id" type="string" optional>
    The universally unique identifier (UUID) of a knowledge base to associate with this evaluation. The knowledge base provides reference data that the evaluation can use to assess responses.
  </ParamField>

  <ParamField body="error_localizer" type="boolean" optional>
    Whether to enable error localization for this evaluation. When `true`, the evaluation engine identifies and highlights specific portions of the output that caused evaluation failures.
  </ParamField>

  <ParamField body="model" type="string" optional>
    The model to use for running the evaluation. This determines which LLM-based evaluation engine processes each row.
  </ParamField>

  <ParamField body="run" type="boolean" optional>
    Whether to re-run the evaluation after updating its configuration. When `true`, all cell statuses are reset to running and rows are queued for reprocessing. When `false`, only the configuration is updated without triggering execution. Default is `false`.
  </ParamField>

  <ParamField body="save_as_template" type="boolean" optional>
    Whether to save the updated configuration as a new reusable eval template. When `true`, the `name` field is required and must be unique across all templates. Default is `false`.
  </ParamField>

  <ParamField body="name" type="string" optional>
    The name for the new eval template. Required when `save_as_template` is `true`. Must be unique across all eval templates in your organization.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="string">

  <ResponseField name="success" type="boolean">
    A boolean flag indicating whether the request completed successfully. Returns `true` when the evaluation is updated without errors.
  </ResponseField>
</ApiSection>
