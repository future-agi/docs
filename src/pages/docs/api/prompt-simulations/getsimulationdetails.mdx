<ApiSection title="Path parameters">
  <ParamField path="prompt_template_id" type="string" required>
    The UUID of the prompt template that the simulation run belongs to. The prompt template must exist within your organization and must not have been deleted.
  </ParamField>
  <ParamField path="run_test_id" type="string" required>
    The UUID of the specific simulation run to retrieve. This must be a valid simulation run that is associated with the specified prompt template. You can retrieve simulation run IDs from the [List simulations](/docs/api/prompt-simulations/listsimulations) endpoint.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="id" type="string">
    The unique UUID identifier of the simulation run.
  </ResponseField>
  <ResponseField name="name" type="string">
    The human-readable name assigned to this simulation run.
  </ResponseField>
  <ResponseField name="description" type="string">
    The description of the simulation run, or `null` if none was provided.
  </ResponseField>
  <ResponseField name="source_type" type="string">
    The internal source type identifier for this simulation (e.g., `"prompt"`).
  </ResponseField>
  <ResponseField name="source_type_display" type="string">
    A human-readable display label for the source type (e.g., `"Prompt"`).
  </ResponseField>
  <ResponseField name="prompt_template" type="string">
    The UUID of the parent prompt template associated with this simulation run.
  </ResponseField>
  <ResponseField name="prompt_template_detail" type="object">
    Detailed information about the parent prompt template, including `id`, `name`, `description`, and `variable_names` (an array of variable placeholders used in the template).
  </ResponseField>
  <ResponseField name="prompt_version" type="string">
    The UUID of the specific prompt version this simulation run is configured to use.
  </ResponseField>
  <ResponseField name="prompt_version_detail" type="object">
    Detailed information about the prompt version, including `id`, `template_version` (integer version number), `is_default` (whether this is the active default version), and `commit_message` (the message provided when this version was committed).
  </ResponseField>
  <ResponseField name="scenarios" type="array of string">
    An array of scenario UUIDs associated with this simulation run.
  </ResponseField>
  <ResponseField name="scenarios_detail" type="array of object">
    An array of detailed scenario objects providing full information about each associated scenario.
  </ResponseField>
  <ResponseField name="dataset_row_ids" type="array of string">
    An array of dataset row UUIDs specifying which data rows are included in this simulation. An empty array indicates all rows from the associated scenarios are used.
  </ResponseField>
  <ResponseField name="simulate_eval_configs" type="array of string">
    An array of evaluation configuration UUIDs attached to this simulation run.
  </ResponseField>
  <ResponseField name="simulate_eval_configs_detail" type="array of object">
    An array of detailed evaluation configuration objects providing full information about each evaluation metric and scoring criteria.
  </ResponseField>
  <ResponseField name="enable_tool_evaluation" type="boolean">
    Indicates whether tool usage evaluation is enabled for this simulation run. When `true`, the evaluation framework assesses tool call correctness during execution.
  </ResponseField>
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp indicating when the simulation run was created.
  </ResponseField>
  <ResponseField name="updated_at" type="string">
    ISO 8601 timestamp indicating when the simulation run was last modified.
  </ResponseField>
  <ResponseField name="last_run_at" type="string">
    ISO 8601 timestamp indicating when the simulation was last executed, or `null` if it has never been run.
  </ResponseField>
</ApiSection>
