    The unique identifier (UUID) of the eval task to retrieve. You can obtain this ID from the response of the [Create Eval Task](/docs/api/eval-tasks/create-eval-task) or [List Eval Tasks](/docs/api/eval-tasks/list-eval-tasks-filtered) endpoints.

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="id" type="string">
    The unique identifier (UUID) of the eval task.
  </ResponseField>
  <ResponseField name="name" type="string">
    The human-readable name assigned to the eval task when it was created.
  </ResponseField>
  <ResponseField name="project" type="string">
    The UUID of the project this eval task is associated with. All spans evaluated by this task belong to this project.
  </ResponseField>
  <ResponseField name="status" type="string">
    The current execution status of the eval task. Possible values are `pending` (queued for execution), `running` (actively evaluating spans), `completed` (finished processing all target spans), `failed` (encountered an unrecoverable error), `paused` (manually paused by a user), or `deleted` (soft-deleted).
  </ResponseField>
  <ResponseField name="sampling_rate" type="number">
    The percentage of eligible spans being evaluated, as a value between `1.0` and `100.0`. For example, `50.0` means half of all matching spans are evaluated.
  </ResponseField>
  <ResponseField name="spans_limit" type="integer">
    The maximum number of spans this task will evaluate. Only present for `historical` run type tasks. For `continuous` tasks, this field is `null`.
  </ResponseField>
  <ResponseField name="run_type" type="string">
    The execution mode of the task. Either `continuous` (evaluates incoming spans indefinitely) or `historical` (evaluates a fixed batch of existing spans up to `spans_limit`).
  </ResponseField>
  <ResponseField name="evals" type="array">
    A list of custom eval config UUIDs attached to this task. These define which evaluations are run on each eligible span.
  </ResponseField>
  <ResponseField name="evals_details" type="array">
    Expanded details for each attached custom eval config, including the config's `id`, `name`, and template information. This provides richer context than the `evals` array of raw UUIDs.
  </ResponseField>
  <ResponseField name="filters" type="object">
    The query filters applied to this task, which narrow which spans are eligible for evaluation. An empty object means all spans in the project are considered.
  </ResponseField>
  <ResponseField name="failed_spans" type="array">
    A list of span IDs that failed during evaluation. Use this to investigate and debug evaluation errors on specific spans.
  </ResponseField>
  <ResponseField name="start_time" type="datetime">
    The ISO 8601 timestamp indicating when the eval task began executing. This is `null` if the task has not started yet.
  </ResponseField>
  <ResponseField name="end_time" type="datetime">
    The ISO 8601 timestamp indicating when the eval task finished executing. Only populated for `completed` or `failed` tasks; `null` for tasks that are still active.
  </ResponseField>
  <ResponseField name="last_run" type="datetime">
    The ISO 8601 timestamp of the most recent evaluation cycle. For `continuous` tasks, this updates with each new batch of spans processed.
  </ResponseField>
  <ResponseField name="created_at" type="datetime">
    The ISO 8601 timestamp of when the eval task was created.
  </ResponseField>
  <ResponseField name="updated_at" type="datetime">
    The ISO 8601 timestamp of the most recent modification to the eval task, including status changes, configuration updates, or evaluation runs.
  </ResponseField>
</ApiSection>
