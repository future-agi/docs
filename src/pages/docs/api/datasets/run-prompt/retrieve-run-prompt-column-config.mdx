    <ApiCollapsible title="Show 1 property">

        <ApiCollapsible title="Show 14 properties">
          <ResponseField name="dataset_id" type="string">
            The universally unique identifier (UUID) of the dataset that contains this run prompt column.
          </ResponseField>

          <ResponseField name="name" type="string">
            The display name of the run prompt column as it appears in the dataset table.
          </ResponseField>

          <ResponseField name="model" type="string">
            The identifier of the language model used for generation (e.g., `gpt-4o`, `claude-3-5-sonnet`).
          </ResponseField>

          <ResponseField name="messages" type="array">
            The ordered array of message objects defining the prompt. Column UUID references within message content are resolved back to human-readable `{{column_name}}` template variable syntax.
          </ResponseField>

          <ResponseField name="temperature" type="number">
            The sampling temperature controlling randomness in the model's output, ranging from `0` to `2`.
          </ResponseField>

          <ResponseField name="frequency_penalty" type="number">
            The frequency penalty applied to tokens based on their occurrence count in the generated text, ranging from `-2` to `2`.
          </ResponseField>

          <ResponseField name="presence_penalty" type="number">
            The presence penalty applied to tokens based on whether they have appeared at all in the generated text, ranging from `-2` to `2`.
          </ResponseField>

          <ResponseField name="max_tokens" type="integer">
            The maximum number of tokens the model is allowed to generate in a single response.
          </ResponseField>

          <ResponseField name="top_p" type="number">
            The nucleus sampling parameter controlling token selection probability mass, ranging from `0` to `1`.
          </ResponseField>

          <ResponseField name="response_format" type="string">
            The response format constraint applied to the model output (e.g., `json_object`), or `null` if no format constraint is set.
          </ResponseField>

          <ResponseField name="tool_choice" type="string">
            The tool selection strategy governing how the model decides whether to call a tool (e.g., `auto`, `none`, `required`), or `null` if no tools are configured.
          </ResponseField>

          <ResponseField name="tools" type="array">
            The array of tool definitions available to the model during generation. Each tool includes an `id`, `name`, and `config` describing its capabilities. Returns an empty array when no tools are configured.
          </ResponseField>

          <ResponseField name="output_format" type="string">
            The output format of the generated content. One of `string` (plain text), `audio` (audio output), or `json` (structured JSON output).
          </ResponseField>

          <ResponseField name="concurrency" type="integer">
            The number of concurrent requests configured for parallel row processing.
          </ResponseField>

          <ResponseField name="run_prompt_config" type="object">
            Additional run prompt settings specific to the column configuration. This object contains provider-specific or advanced configuration options.
          </ResponseField>
        </ApiCollapsible>
