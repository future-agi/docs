<ApiSection title="Request body">
  <ParamField body="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset from which to retrieve sample column values. The dataset must exist within your organization and be accessible with your current API credentials.
  </ParamField>

  <ParamField body="column_placeholders" type="object" required>
    A mapping of placeholder names to column UUIDs. The keys represent the placeholder variable names used in prompt templates (e.g., `input`, `context`), and the values are the UUIDs of the corresponding dataset columns. This allows you to preview what data will be injected into template variables before configuring a run prompt column.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the outcome of the request. Returns `success` when column values are retrieved without errors.
  </ResponseField>

  <ResponseField name="data" type="object">
    The response payload containing the retrieved column values organized by placeholder name.

  </ResponseField>
    <ApiCollapsible title="Show 1 property">
      <ResponseField name="result" type="object">
        An object keyed by placeholder name, where each entry contains the column metadata and sample values. Up to 10 rows of data are returned for each requested column.

      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="column_id" type="string">
            The universally unique identifier (UUID) of the column from which the sample values were retrieved.
          </ResponseField>

          <ResponseField name="column_name" type="string">
            The display name of the column in the dataset, as it appears in the table header.
          </ResponseField>

          <ResponseField name="values" type="array">
            An array of sample values from the column, containing up to 10 entries from the first rows of the dataset. Useful for previewing what data will populate template variables at runtime.
          </ResponseField>
        </ApiCollapsible>
    </ApiCollapsible>
</ApiSection>
