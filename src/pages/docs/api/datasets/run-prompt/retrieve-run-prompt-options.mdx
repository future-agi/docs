<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the outcome of the request. Returns `success` when the available options are retrieved without errors.
  </ResponseField>

  <ResponseField name="data" type="object">
    The response payload containing all available configuration options for setting up a run prompt column, including models, tools, output formats, and tool selection strategies.

  </ResponseField>
    <ApiCollapsible title="Show 5 properties">
      <ResponseField name="models" type="array">
        An array of available language model objects that can be used for run prompt generation. Each model entry indicates its name, supported providers, and current availability status.

      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="model_name" type="string">
            The identifier of the language model (e.g., `gpt-4o`, `claude-3-5-sonnet`). Use this value in the `config.model` field when creating or editing a run prompt column.
          </ResponseField>

          <ResponseField name="providers" type="array">
            An array of provider names that offer this model (e.g., `["openai"]`, `["anthropic"]`). A model may be available from multiple providers.
          </ResponseField>

          <ResponseField name="is_available" type="boolean">
            Indicates whether the model is currently available for use. Models may become temporarily unavailable due to provider outages or capacity constraints.
          </ResponseField>
        </ApiCollapsible>

      <ResponseField name="tool_config" type="object">
        The tool configuration schema describing the structure and validation rules for tool definitions. This schema guides how tools should be configured when added to a run prompt column.
      </ResponseField>

      <ResponseField name="available_tools" type="array">
        An array of tool objects available to your organization for use in run prompt columns. These tools can be referenced in the `config.tools` array when creating or editing a run prompt column.

      </ResponseField>
        <ApiCollapsible title="Show 5 properties">
          <ResponseField name="id" type="string">
            The universally unique identifier (UUID) of the tool. Use this value when referencing the tool in a run prompt column configuration.
          </ResponseField>

          <ResponseField name="name" type="string">
            The human-readable name of the tool, describing its function (e.g., `web_search`).
          </ResponseField>

          <ResponseField name="config" type="object">
            The configuration object for the tool, containing provider-specific settings and parameters.
          </ResponseField>

          <ResponseField name="config_type" type="string">
            The type of tool configuration, indicating how the tool is invoked by the model (e.g., `function`).
          </ResponseField>

          <ResponseField name="description" type="string">
            A human-readable description of what the tool does and when the model should use it.
          </ResponseField>
        </ApiCollapsible>

      <ResponseField name="output_formats" type="array">
        An array of supported output format options, each containing a `value` (the programmatic identifier to use in configuration) and a `label` (the human-readable display name). Standard options include `string`, `audio`, and `json`.
      </ResponseField>

      <ResponseField name="tool_choices" type="array">
        An array of supported tool choice strategy options, each containing a `value` (the programmatic identifier to use in configuration) and a `label` (the human-readable display name). Standard options include `auto`, `none`, and `required`.
      </ResponseField>
    </ApiCollapsible>
</ApiSection>
