<ApiSection title="Query parameters">
  <ParamField query="model" type="string" required>
    The identifier of the language model whose audio voice options you want to retrieve (e.g., `gpt-4o-audio-preview`). The model must support audio output capabilities. This parameter determines which provider's voice catalog is returned.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the outcome of the request. Returns `success` when voice options are retrieved without errors.
  </ResponseField>

  <ResponseField name="data" type="object">
    The voice configuration payload for the specified model, including available voices, supported audio formats, and default settings.

  </ResponseField>
    <ApiCollapsible title="Show 6 properties">
      <ResponseField name="model_name" type="string">
        The identifier of the model for which voice options were retrieved, matching the value provided in the `model` query parameter.
      </ResponseField>

      <ResponseField name="provider" type="string">
        The name of the model provider that serves this model (e.g., `openai`, `elevenlabs`). This determines which voice catalog and audio rendering engine is used.
      </ResponseField>

      <ResponseField name="custom_voice_supported" type="boolean">
        Indicates whether this model supports custom (user-created) voices in addition to the built-in system voices. When `true`, custom TTS voices created by your organization can be used with this model.
      </ResponseField>

      <ResponseField name="supported_voices" type="array">
        An array of voice objects representing all voices available for use with this model. Each voice includes its unique identifier, display name, and whether it is a built-in system voice or a custom voice created by your organization.

      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="id" type="string">
            The unique identifier of the voice, used when configuring audio output in a run prompt column.
          </ResponseField>

          <ResponseField name="name" type="string">
            The human-readable display name of the voice, shown in the UI when selecting voice options.
          </ResponseField>

          <ResponseField name="type" type="string">
            The category of the voice. `system` indicates a built-in voice provided by the model provider, while `custom` indicates a voice created by your organization.
          </ResponseField>
        </ApiCollapsible>

      <ResponseField name="supported_formats" type="array">
        An array of strings listing the audio file formats supported by this model (e.g., `mp3`, `wav`, `opus`, `flac`). These determine which output encoding options are available when generating audio responses.
      </ResponseField>

      <ResponseField name="default_voice" type="string">
        The identifier of the default voice used when no specific voice is selected in the prompt configuration. This voice is automatically applied if the `voice` parameter is omitted.
      </ResponseField>

      <ResponseField name="default_format" type="string">
        The default audio file format used when no specific format is selected in the prompt configuration (e.g., `mp3`). This format is automatically applied if the `format` parameter is omitted.
      </ResponseField>
    </ApiCollapsible>
</ApiSection>
