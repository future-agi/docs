<ApiSection title="Path parameters">
  <ParamField path="id" type="string" required>
    The universally unique identifier (UUID) of the target dataset to which empty rows will be appended. The dataset must exist within your organization and can have any number of existing columns and rows.
  </ParamField>
</ApiSection>

<ApiSection title="Request body">
  <ParamField body="num_rows" type="integer" required>
    The number of empty rows to add to the dataset. Each new row will have blank values for all existing columns. The value must be an integer between 1 and 100, inclusive. Use this to pre-allocate rows that you intend to populate later via the Update Cell Value endpoint or through the dashboard.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="string">

  </ResponseField>
  <ResponseField name="status" type="string">
    Indicates the outcome of the API request. Returns `"success"` when the empty rows are added without errors.
  </ResponseField>
</ApiSection>
