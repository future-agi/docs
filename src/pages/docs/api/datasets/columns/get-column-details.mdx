<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset whose column metadata you want to retrieve. This ID is returned when you create a dataset and can be found in the dataset listing or dataset detail endpoints.
  </ParamField>
</ApiSection>

<ApiSection title="Query parameters">
  <ParamField query="include_prompt" type="boolean" optional>
    Whether to include columns with RUN_PROMPT source type in the response. By default, RUN_PROMPT columns are excluded from the results. Set to `true` to include them alongside static and other column types.
  </ParamField>

  <ParamField query="source" type="string" optional>
    Filter the returned columns by their source type. When specified, only columns matching the given source are returned. Common values include `STATIC` for manually created columns, `RUN_PROMPT` for prompt-generated columns, and `EVALUATION` for evaluation metric columns.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the result of the request. Returns `success` when the column details are retrieved successfully.
  </ResponseField>

  <ResponseField name="data" type="object">
    The response payload containing the column configuration for the specified dataset.
  </ResponseField>
    <ApiCollapsible title="Show 1 property">
      <ResponseField name="column_config" type="array">
        An ordered list of column metadata objects representing all columns in the dataset that match the applied filters. Each object provides the essential identifiers and type information for a single column.
      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="id" type="string">
            The universally unique identifier (UUID) assigned to the column. Use this ID to reference the column in subsequent API calls such as updating column names, changing column types, or deleting columns.
          </ResponseField>
          <ResponseField name="name" type="string">
            The human-readable name of the column as it appears in the dataset grid and throughout the platform.
          </ResponseField>
          <ResponseField name="data_type" type="string">
            The data type of the column, which determines what kind of values can be stored in it. One of: `text`, `number`, `boolean`, `json`, `image`, `audio`, `pdf`.
          </ResponseField>
        </ApiCollapsible>
    </ApiCollapsible>
