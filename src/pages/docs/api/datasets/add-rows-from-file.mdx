<ApiSection title="Request body">
  <ParamField body="file" type="file" required>
    The file to upload containing the row data to be added to the dataset. Supported formats include `.csv`, `.xls`, `.xlsx`, `.json`, and `.jsonl`. The maximum allowed file size is 10 MB. The request must use `multipart/form-data` content type when uploading this field. Column headers in the file are matched to existing dataset columns by name.
  </ParamField>
  <ParamField body="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the target dataset to which the uploaded rows will be added. The dataset must exist within your organization and must have columns that correspond to the file's column headers or keys.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="data" type="string">

  </ResponseField>
  <ResponseField name="status" type="string">
    Indicates the outcome of the API request. Returns `"success"` when the file upload and row import completes without errors.
  </ResponseField>
</ApiSection>
