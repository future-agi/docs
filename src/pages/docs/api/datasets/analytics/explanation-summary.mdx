<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset for which you want to retrieve or generate an AI-powered explanation summary. This ID is returned when you create a dataset and can be found in the dataset listing or dataset detail endpoints. If no summary has been generated yet, calling this endpoint automatically triggers a background generation task.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the result of the request. Returns `success` when the explanation summary data is retrieved, regardless of whether the summary itself has been generated yet.
  </ResponseField>

  <ResponseField name="data" type="object">
    The explanation summary data and generation status. The `response` field contains the actual summary content when available, while the `status` field indicates whether the summary has been generated, is in progress, or cannot be generated due to insufficient data.
  </ResponseField>
    <ApiCollapsible title="Show 5 properties">
      <ResponseField name="response" type="object">
        The AI-generated explanation summary content. Contains structured insights about the dataset's content, patterns, and potential issues. Returns `null` if the summary has not yet been generated or if the dataset has insufficient data.
      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="summary" type="string">
            A comprehensive natural-language summary describing the dataset's content, structure, and overall characteristics. Provides a high-level overview of what the data represents and its key attributes.
          </ResponseField>
          <ResponseField name="key_patterns" type="array">
            A list of notable patterns, trends, and correlations identified by the AI analysis across the dataset's rows and columns. Each entry is a string describing a specific observation.
          </ResponseField>
          <ResponseField name="critical_issues" type="array">
            A list of potential quality issues, anomalies, or concerns identified in the dataset. Each entry is a string describing a specific issue that may require attention, such as data inconsistencies, hallucinations, or coverage gaps.
          </ResponseField>
        </ApiCollapsible>
      <ResponseField name="last_updated" type="string">
        The ISO 8601 timestamp indicating when the explanation summary was last generated or refreshed. Returns `null` if the summary has never been generated.
      </ResponseField>
      <ResponseField name="status" type="string">
        The current generation status of the explanation summary. Possible values: `COMPLETED` indicates the summary is available and ready to read; `PENDING` indicates the summary generation is currently in progress as a background task; `INSUFFICIENT_DATA` indicates the dataset does not have enough rows to generate a meaningful summary.
      </ResponseField>
      <ResponseField name="row_count" type="integer">
        The current number of rows in the dataset at the time of the request. Used in conjunction with `min_rows_required` to determine whether the dataset has sufficient data for summary generation.
      </ResponseField>
      <ResponseField name="min_rows_required" type="integer">
        The minimum number of rows required in the dataset before an explanation summary can be generated. If `row_count` is less than this value, the status will be `INSUFFICIENT_DATA`.
      </ResponseField>
    </ApiCollapsible>
</ApiSection>
