<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the result of the request. Returns `success` when the annotation summary is retrieved successfully.
  </ResponseField>

  <ResponseField name="data" type="object">
    The aggregated annotation summary for the dataset, including overall progress metrics and the distribution of annotation labels applied across all rows.
  </ResponseField>
    <ApiCollapsible title="Show 5 properties">
      <ResponseField name="total_annotations" type="integer">
        The total number of individual annotations that have been applied across all rows in the dataset. A single row may have multiple annotations if reviewed by multiple annotators or annotated multiple times.
      </ResponseField>
      <ResponseField name="total_rows" type="integer">
        The total number of rows in the dataset, regardless of whether they have been annotated. This provides the denominator for calculating annotation coverage.
      </ResponseField>
      <ResponseField name="annotated_rows" type="integer">
        The number of rows in the dataset that have received at least one annotation. Combined with `total_rows`, this indicates how much of the dataset has been reviewed.
      </ResponseField>
      <ResponseField name="annotation_coverage" type="number">
        The ratio of annotated rows to total rows, expressed as a decimal between 0 and 1. Calculated as `annotated_rows / total_rows`. Provides a quick measure of how complete the annotation effort is for this dataset.
      </ResponseField>
      <ResponseField name="labels" type="array">
        A breakdown of annotation labels applied across the dataset, showing the distribution of each label. Useful for understanding the balance of categories in your annotated data.
      </ResponseField>
        <ApiCollapsible title="Show 3 properties">
          <ResponseField name="label_name" type="string">
            The display name of the annotation label as defined in your annotation configuration. Examples include `Correct`, `Incorrect`, `Partially Correct`, or any custom labels you have configured.
          </ResponseField>
          <ResponseField name="count" type="integer">
            The total number of times this specific label has been applied across all annotations in the dataset.
          </ResponseField>
          <ResponseField name="percentage" type="number">
            The proportion of total annotations that used this label, expressed as a decimal between 0 and 1. Calculated as `count / total_annotations`.
          </ResponseField>
        </ApiCollapsible>
    </ApiCollapsible>
</ApiSection>
