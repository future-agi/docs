<ApiSection title="Authentication">
  <ParamField name="X-Api-Key" type="string" required>
    Your Future AGI API key used to authenticate requests. You can find and manage your API keys in the [Dashboard](https://app.futureagi.com) under Settings.
  </ParamField>
  <ParamField name="X-Secret-Key" type="string" required>
    Your Future AGI secret key, used alongside the API key for request authentication. This is generated when you create an API key in the [Dashboard](https://app.futureagi.com).
  </ParamField>
</ApiSection>

<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset for which you want to retrieve run prompt statistics. This ID is returned when you create a dataset and can be found in the dataset listing or dataset detail endpoints.
  </ParamField>
</ApiSection>

<ApiSection title="Query parameters">
  <ParamField query="prompt_ids" type="string" optional>
    A comma-separated list of RunPrompter UUIDs to filter the returned statistics. When provided, only statistics for the specified run prompt columns are included in the response. When omitted, statistics for all run prompt columns in the dataset are returned.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A status message indicating the result of the request. Returns `success` when the statistics are retrieved successfully.
  </ResponseField>

  <ResponseField name="data" type="object">
    The aggregated run prompt statistics for the dataset, including overall averages and per-prompt breakdowns.
  </ResponseField>
    <ApiCollapsible title="Show 4 properties">
      <ResponseField name="avg_tokens" type="number">
        The average number of tokens consumed per prompt execution across all run prompt columns in the dataset. This includes both input and output tokens and provides an overall measure of token usage efficiency.
      </ResponseField>
      <ResponseField name="avg_cost" type="number">
        The average cost in USD per prompt execution across all run prompt columns in the dataset. Calculated based on the model pricing for the tokens consumed during each execution.
      </ResponseField>
      <ResponseField name="avg_time" type="number">
        The average response time in seconds per prompt execution across all run prompt columns in the dataset. Measures the end-to-end latency from sending the prompt to receiving the complete response from the language model.
      </ResponseField>
      <ResponseField name="prompts" type="array">
        A detailed per-prompt breakdown of execution statistics. Each object in the array represents a single run prompt column and its individual performance metrics.
      </ResponseField>
        <ApiCollapsible title="Show 9 properties">
          <ResponseField name="id" type="string">
            The universally unique identifier (UUID) of the RunPrompter configuration associated with this run prompt column.
          </ResponseField>
          <ResponseField name="name" type="string">
            The human-readable name of the run prompt column as it appears in the dataset grid.
          </ResponseField>
          <ResponseField name="model" type="string">
            The identifier of the language model used for prompt execution in this column, such as `gpt-4o`, `gpt-3.5-turbo`, or `claude-3-opus`.
          </ResponseField>
          <ResponseField name="avg_tokens" type="number">
            The average number of tokens consumed per execution for this specific run prompt column.
          </ResponseField>
          <ResponseField name="avg_cost" type="number">
            The average cost in USD per execution for this specific run prompt column.
          </ResponseField>
          <ResponseField name="avg_time" type="number">
            The average response time in seconds per execution for this specific run prompt column.
          </ResponseField>
          <ResponseField name="total_rows" type="integer">
            The total number of rows in the dataset that this run prompt column was configured to process.
          </ResponseField>
          <ResponseField name="completed_rows" type="integer">
            The number of rows that were successfully processed by this run prompt column, producing a valid output.
          </ResponseField>
          <ResponseField name="failed_rows" type="integer">
            The number of rows where prompt execution failed for this column, due to errors such as model timeouts, rate limits, or invalid input data.
          </ResponseField>
        </ApiCollapsible>
    </ApiCollapsible>
</ApiSection>
