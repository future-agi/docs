<ApiSection title="Path parameters">
  <ParamField path="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the dataset for which you want to retrieve evaluation statistics. This ID is returned when you create a dataset and can be found in the dataset listing or dataset detail endpoints.
  </ParamField>

<ApiSection title="Query parameters">
  <ParamField query="column_ids" type="string" optional>
    A comma-separated list of evaluation column UUIDs to filter the returned statistics. When provided, only statistics for the specified columns are included in the response. When omitted, statistics for all evaluation columns in the dataset are returned.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  Returns an array of evaluation statistics objects, one per evaluation template applied to the dataset. Each object provides aggregated metrics summarizing the evaluation results.

  <ResponseField name="template_name" type="string">
    The name of the evaluation template that produced these metrics. This corresponds to the template configured on the evaluation column, such as `Relevance`, `Faithfulness`, `Coherence`, or a custom template name.
  </ResponseField>

  <ResponseField name="metric_count" type="integer">
    The total number of data points (rows) that were evaluated by this template. This count includes both passed and failed evaluations.
  </ResponseField>

  <ResponseField name="average_score" type="number">
    The arithmetic mean of all evaluation scores produced by this template across the dataset. Scores are typically normalized to a 0-to-1 range, where higher values indicate better performance.
  </ResponseField>

  <ResponseField name="min_score" type="number">
    The lowest evaluation score recorded for this template across all evaluated rows in the dataset. Useful for identifying worst-case performance.
  </ResponseField>

  <ResponseField name="max_score" type="number">
    The highest evaluation score recorded for this template across all evaluated rows in the dataset. Useful for understanding the upper bound of performance.
  </ResponseField>

  <ResponseField name="pass_count" type="integer">
    The number of evaluated rows that met or exceeded the passing threshold defined by the evaluation template. Combined with `fail_count`, this accounts for all evaluated rows.
  </ResponseField>

  <ResponseField name="fail_count" type="integer">
    The number of evaluated rows that fell below the passing threshold defined by the evaluation template. Indicates how many data points did not meet the quality criteria.
  </ResponseField>

  <ResponseField name="pass_rate" type="number">
    The ratio of passed evaluations to total evaluations, expressed as a decimal between 0 and 1. Calculated as `pass_count / metric_count`. Provides a quick summary of overall evaluation quality.
  </ResponseField>
</ApiSection>
