<ApiSection title="Authentication">
  <ParamField name="X-Api-Key" type="API Key" required>

<ApiSection title="Request body">
  <ParamField body="dataset_id" type="string" required>
    The universally unique identifier (UUID) of the source dataset or experiment table from which columns and rows will be copied. The source must exist within your organization. All rows from the source dataset will be included in the new dataset, but only the columns specified in the `columns` mapping will be carried over.
  </ParamField>
  <ParamField body="name" type="string" required>
    The name for the newly created dataset. This name must be unique within your organization. Choose a descriptive name that helps identify the dataset's purpose or contents for easy reference later.
  </ParamField>
  <ParamField body="columns" type="object" required>
    A JSON object that maps source column UUIDs to the desired column names in the new dataset. Each key is a UUID of a column in the source dataset, and each value is the string name that column should have in the new dataset. At least one column mapping is required, and no duplicate column names are allowed. Format: `{ "source_column_id": "new_column_name" }`.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="message" type="string">
    A human-readable confirmation message indicating the new dataset was created successfully, such as `"Dataset created successfully"`.
  </ResponseField>
  <ResponseField name="dataset_id" type="string">
    The universally unique identifier (UUID) of the newly created dataset. Use this ID to reference the dataset in subsequent API calls, such as adding rows, running evaluations, or retrieving dataset details.
  </ResponseField>
</ApiSection>
