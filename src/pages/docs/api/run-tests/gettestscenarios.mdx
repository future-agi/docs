<ApiSection title="Path parameters">
  <ParamField path="run_test_id" type="string" required>
    The UUID of the test run whose associated scenarios you want to retrieve. This identifier is returned when you create a test run via the [Create Test Run](/docs/api/run-tests/createruntest) endpoint, or can be found in the results from the [List Test Runs](/docs/api/run-tests/listruntests) endpoint.
  </ParamField>
</ApiSection>

<ApiSection title="Query parameters">
  <ParamField query="search" type="string" optional>
    A case-insensitive search string used to filter scenarios by their name, source, or type. The search performs a partial match, so searching for "checkout" will return scenarios with "Checkout Flow" in their name. Leave empty to return all scenarios associated with the test run.
  </ParamField>
  <ParamField query="limit" type="integer" optional>
    The maximum number of scenario records to return per page. Controls the page size for paginated results. Defaults to `10` if not specified. Must be a positive integer. Use higher values when you need to retrieve all scenarios in fewer requests.
  </ParamField>
  <ParamField query="page" type="integer" optional>
    The page number to retrieve in the paginated result set. Defaults to `1` (the first page). Use this in combination with `limit` to navigate through the full list of scenarios when a test run has many associated scenarios.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">

  <ResponseField name="previous" type="string or null">
    A fully-qualified URL to the previous page of results. Returns `null` when the current page is the first page.
  </ResponseField>
  <ResponseField name="results" type="array of objects">
    An array of scenario summary objects for the current page.
  </ResponseField>
  <ApiCollapsible title="Show 3 properties">
    <ResponseField name="id" type="string">

    <ResponseField name="name" type="string">

    <ResponseField name="row_count" type="integer">
      The total number of data rows contained within this scenario. Each row represents a distinct test case variation that the agent will be tested against during execution. A higher row count means more individual calls will be generated when this scenario is executed.
