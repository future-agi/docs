<ApiSection title="Path parameters">
  <ParamField path="run_test_id" type="string" required>
    The universally unique identifier (UUID) of the test run whose evaluation summary you want to retrieve. This ID is returned when you create a test run via the API or can be found in the Future AGI Dashboard under the test run details page.
  </ParamField>
</ApiSection>

<ApiSection title="Query parameters">
  <ParamField query="execution_id" type="string">
    An optional UUID of a specific test execution within the test run. When provided, the response is filtered to return evaluation summary statistics only for that particular execution rather than the aggregated summary across all executions. If omitted, the summary covers all executions in the test run.
  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="evaluations" type="array">
    An array of evaluation summary objects, one per evaluation configuration that was run against the test.
  </ResponseField>
  <ApiCollapsible title="Show 5 properties">
    <ResponseField name="name" type="string">
      The name of the evaluation configuration.
    </ResponseField>
    <ResponseField name="average_score" type="number">
      The average score across all evaluated call executions for this eval config, on a scale determined by the evaluation template.
    </ResponseField>
    <ResponseField name="total_runs" type="integer">
      The total number of evaluation runs performed for this config across all calls in the test.
    </ResponseField>
    <ResponseField name="passed" type="integer">
      The number of evaluations that passed for this config.
    </ResponseField>
    <ResponseField name="failed" type="integer">
      The number of evaluations that failed for this config.
    </ResponseField>
  </ApiCollapsible>
</ApiSection>
