<ApiSection title="Query parameters">
  <ParamField query="source_type" type="string">

  </ParamField>
</ApiSection>

<ApiSection title="Response" status={200} statusText="OK">
  An array of queue entries the current user can annotate. Each entry has:
  <ResponseField name="queue" type="object">Queue summary with `id`, `name`, `instructions`, `is_default`.</ResponseField>
  <ResponseField name="item" type="object">The queue item for this source, or null if none exists. Contains `id`, `status`, `source_type`.</ResponseField>
  <ResponseField name="labels" type="array">Labels attached to the queue.</ResponseField>
  <ResponseField name="existing_scores" type="object">Map of label UUID → current user's annotation value for this source.</ResponseField>
  <ResponseField name="existing_notes" type="string">Current user's existing note for this source.</ResponseField>
  <ResponseField name="existing_label_notes" type="object">Map of label UUID → per-label note text.</ResponseField>
  <ResponseField name="span_notes" type="array">All span notes for observation_span sources. Each has `id`, `notes`, `annotator`, `created_at`.</ResponseField>
</ApiSection>
