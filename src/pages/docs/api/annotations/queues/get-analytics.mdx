<ApiSection title="Response" status={200} statusText="OK">
  <ResponseField name="throughput" type="object">
    Throughput metrics. Contains `daily` (array of `{date, count}` for the last 30 days), `total_completed`, and `avg_per_day`.
  </ResponseField>
  <ResponseField name="annotator_performance" type="array">
    Per-annotator performance. Each entry has `user_id`, `name`, `completed`, `last_active`.
  </ResponseField>
  <ResponseField name="label_distribution" type="object">
    Per-label value distribution keyed by label UUID. Each entry has `name`, `type`, and `values` (map of value → count).
  </ResponseField>
  <ResponseField name="status_breakdown" type="object">
    Item counts by status (e.g. `{"pending": 50, "in_progress": 8, "completed": 42}`).
  </ResponseField>
  <ResponseField name="total" type="integer">Total number of items in the queue.</ResponseField>
</ApiSection>
