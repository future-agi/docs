### File Upload Configuration

```python
from fi.datasets import Dataset
from fi.datasets.upload import FileUploader
from fi.datasets.types import UploadConfig, DataTypeChoices

# Configure file upload
upload_config = UploadConfig(
    file_format="csv",
    delimiter=",",
    header_row=True,
    encoding="utf-8",
    validation_level="strict",
    auto_detect_types=True
)

# Initialize uploader
uploader = FileUploader(config=upload_config)

# Upload and create dataset
dataset = uploader.upload_file(
    file_path="customer_data.csv",
    dataset_name="customer_feedback_analysis",
    description="Customer feedback data for analysis"
)
```

## SDK Integration for File Upload

For programmatic file uploads, use the enhanced SDK:

```python
from fi.datasets import Dataset
from fi.datasets.upload import FileUploader
from fi.datasets.types import UploadConfig, ValidationLevel

# Configure advanced upload settings
upload_config = UploadConfig(
    validation_level=ValidationLevel.STRICT,
    auto_detect_types=True,
    handle_duplicates="skip",
    batch_size=1000,
    parallel_processing=True,
    error_threshold=0.05  # Allow 5% error rate
)

# Initialize uploader with configuration
uploader = FileUploader(config=upload_config)

try:
    # Upload file with progress tracking
    result = uploader.upload_file(
        file_path="large_dataset.csv",
        dataset_name="production_dataset",
        description="Production data for model training",
        on_progress=lambda progress: print(f"Upload progress: {progress}%"),
        on_error=lambda error: print(f"Upload error: {error}")
    )
    
    print(f"Upload completed successfully!")
    print(f"Dataset ID: {result.dataset_id}")
    print(f"Rows processed: {result.rows_processed}")
    print(f"Errors encountered: {result.error_count}")
    
except Exception as e:
    print(f"Upload failed: {e}")
```
