### **4. SDK Integration for Hugging Face Import**

For programmatic access, use the enhanced SDK to import Hugging Face datasets:

```python
from fi.datasets import Dataset
from fi.datasets.types import DatasetConfig, HuggingfaceDatasetConfig, ModelTypes

# Configure dataset
dataset_config = DatasetConfig(
    name="squad_qa_dataset",
    model_type=ModelTypes.GENERATIVE_LLM
)

# Configure Hugging Face dataset
hf_config = HuggingfaceDatasetConfig(
    name="squad",  # Hugging Face dataset name
    subset="v2.0",  # Optional: specify subset (e.g., v1.1, v2.0 for SQuAD)
    split="train",  # Optional: specify split (train, validation, test)
    num_rows=100   # Optional: limit number of rows to import
)

# Create dataset from Hugging Face
try:
    dataset = Dataset.create_dataset(
        dataset_config=dataset_config,
        source=hf_config
    )
    print(f"✓ Dataset created successfully: {dataset.dataset_config.name}")
    print(f"  Dataset ID: {dataset.dataset_config.id}")
    
    # Download and verify the data
    df = dataset.download(load_to_pandas=True)
    print(f"  Shape: {df.shape}")
    print(f"  Columns: {list(df.columns)}")
    
except Exception as e:
    print(f"✗ Error creating dataset: {e}")
```

### **5. Advanced SDK Usage Examples**

#### **Import with Custom Configuration**

```python
import os
from fi.datasets import Dataset
from fi.datasets.types import DatasetConfig, HuggingfaceDatasetConfig, ModelTypes

# Set up environment
os.environ["FI_API_KEY"] = "your_api_key"
os.environ["FI_SECRET_KEY"] = "your_secret_key"

def import_hf_dataset():
    """Complete example of importing a Hugging Face dataset"""
    
    # Configure dataset
    config = DatasetConfig(
        name="imdb_sentiment_analysis",
        model_type=ModelTypes.CLASSIFICATION
    )
    
    # Configure Hugging Face import
    hf_config = HuggingfaceDatasetConfig(
        name="imdb",
        split="train",
        num_rows=1000  # Import first 1000 rows
    )
    
    try:
        # Create dataset
        dataset = Dataset.create_dataset(
            dataset_config=config,
            source=hf_config
        )
        
        print(f"✓ Successfully imported IMDB dataset")
        print(f"  Dataset Name: {dataset.dataset_config.name}")
        print(f"  Dataset ID: {dataset.dataset_config.id}")
        
        # Verify import
        data = dataset.download(load_to_pandas=True)
        print(f"  Imported {len(data)} rows")
        print(f"  Columns: {list(data.columns)}")
        
        return dataset
        
    except Exception as e:
        print(f"✗ Import failed: {e}")
        return None

# Run the import
if __name__ == "__main__":
    dataset = import_hf_dataset()
```

#### **Batch Import Multiple Datasets**

```python
def import_multiple_hf_datasets():
    """Import multiple Hugging Face datasets"""
    
    datasets_to_import = [
        {
            "name": "squad_v1",
            "hf_name": "squad",
            "subset": "v1.1",
            "split": "train",
            "num_rows": 500
        },
        {
            "name": "squad_v2", 
            "hf_name": "squad",
            "subset": "v2.0",
            "split": "train",
            "num_rows": 500
        }
    ]
    
    imported_datasets = []
    
    for dataset_info in datasets_to_import:
        try:
            # Configure dataset
            config = DatasetConfig(
                name=dataset_info["name"],
                model_type=ModelTypes.GENERATIVE_LLM
            )
            
            # Configure Hugging Face import
            hf_config = HuggingfaceDatasetConfig(
                name=dataset_info["hf_name"],
                subset=dataset_info.get("subset"),
                split=dataset_info["split"],
                num_rows=dataset_info["num_rows"]
            )
            
            # Import dataset
            dataset = Dataset.create_dataset(
                dataset_config=config,
                source=hf_config
            )
            
            imported_datasets.append(dataset)
            print(f"✓ Imported {dataset_info['name']}")
            
        except Exception as e:
            print(f"✗ Failed to import {dataset_info['name']}: {e}")
    
    return imported_datasets

# Run batch import
imported = import_multiple_hf_datasets()
print(f"Successfully imported {len(imported)} datasets")
```

#### **Best Practices for SDK Import**

1. **Start Small**: Begin with a small number of rows to test the import
2. **Verify Credentials**: Ensure API credentials are properly set
3. **Check Dataset Availability**: Verify the dataset exists on Hugging Face
4. **Handle Errors Gracefully**: Implement proper error handling
5. **Monitor Progress**: For large imports, monitor progress and handle timeouts
6. **Validate Results**: Always verify the imported data structure and content

#### **Error Handling Template**

```python
def robust_hf_import(dataset_name, hf_dataset_name, **kwargs):
    """Robust Hugging Face dataset import with error handling"""
    
    try:
        # Validate environment
        if not os.getenv("FI_API_KEY") or not os.getenv("FI_SECRET_KEY"):
            raise ValueError("API credentials not set")
        
        # Configure dataset
        config = DatasetConfig(
            name=dataset_name,
            model_type=ModelTypes.GENERATIVE_LLM
        )
        
        # Configure Hugging Face import
        hf_config = HuggingfaceDatasetConfig(
            name=hf_dataset_name,
            **kwargs
        )
        
        # Import with retry logic
        max_retries = 3
        for attempt in range(max_retries):
            try:
                dataset = Dataset.create_dataset(
                    dataset_config=config,
                    source=hf_config
                )
                print(f"✓ Import successful on attempt {attempt + 1}")
                return dataset
                
            except Exception as e:
                if attempt < max_retries - 1:
                    print(f"Attempt {attempt + 1} failed, retrying...")
                    time.sleep(2 ** attempt)  # Exponential backoff
                else:
                    raise e
                    
    except Exception as e:
        print(f"✗ Import failed: {e}")
        return None

# Usage
dataset = robust_hf_import(
    dataset_name="my_squad_dataset",
    hf_dataset_name="squad",
    subset="v2.0",
    split="train",
    num_rows=100
)
```
