## **1. Installation and Setup**

Before proceeding, ensure that the Future AGI SDK is installed and properly configured.

**Installation**

```bash
pip install futureagi
```

**Set up API credentials**

Set your API credentials as environment variables:

```bash
export FI_API_KEY="your_api_key"
export FI_SECRET_KEY="your_secret_key"
```

Or set them programmatically:

```python
import os
os.environ["FI_API_KEY"] = "your_api_key"
os.environ["FI_SECRET_KEY"] = "your_secret_key"
```

## **2. Creating a Dataset**

### **Initialize the Dataset**

The `Dataset` class manages dataset creation and operations. First, define the dataset properties:

```python
from fi.datasets import Dataset
from fi.datasets.types import (
    DatasetConfig, Column, Row, Cell, 
    DataTypeChoices, SourceChoices, ModelTypes
)

# Define dataset configuration
dataset_config = DatasetConfig(
    name="my_dataset",
    model_type=ModelTypes.GENERATIVE_LLM
)

# Initialize dataset instance
dataset = Dataset(dataset_config=dataset_config)
```

### **Create the Dataset**

```python
# Create the dataset
dataset = dataset.create()
print(f"Dataset created: {dataset.dataset_config.name}")
```

## **3. Adding Data Structure**

### **Define Columns**

Create columns with proper data types and sources:

```python
# Define columns with enhanced type safety
columns = [
    Column(
        name="user_query",
        data_type=DataTypeChoices.TEXT,
        source=SourceChoices.OTHERS
    ),
    Column(
        name="response_quality",
        data_type=DataTypeChoices.INTEGER,
        source=SourceChoices.OTHERS
    ),
    Column(
        name="is_helpful",
        data_type=DataTypeChoices.BOOLEAN,
        source=SourceChoices.OTHERS
    ),
    Column(
        name="metadata",
        data_type=DataTypeChoices.JSON,
        source=SourceChoices.OTHERS
    )
]

# Add columns to dataset
dataset = dataset.add_columns(columns=columns)
print("Columns added successfully")
```

### **Add Data Rows**

Create structured rows with cells:

```python
# Define rows with structured data
rows = [
    Row(
        order=1,
        cells=[
            Cell(column_name="user_query", value="What is machine learning?"),
            Cell(column_name="response_quality", value=8),
            Cell(column_name="is_helpful", value=True),
            Cell(column_name="metadata", value='{"model": "gpt-4", "tokens": 150}')
        ]
    ),
    Row(
        order=2,
        cells=[
            Cell(column_name="user_query", value="How does AI work?"),
            Cell(column_name="response_quality", value=9),
            Cell(column_name="is_helpful", value=True),
            Cell(column_name="metadata", value='{"model": "gpt-4", "tokens": 200}')
        ]
    )
]

# Add rows to dataset
dataset = dataset.add_rows(rows=rows)
print("Rows added successfully")
```

## **4. Alternative: Using Class Methods**

For simple operations, you can use class methods:

```python
# Add columns using class method
columns_dict = [
    {
        "name": "text",
        "data_type": DataTypeChoices.TEXT,
        "source": SourceChoices.OTHERS
    },
    {
        "name": "label", 
        "data_type": DataTypeChoices.INTEGER,
        "source": SourceChoices.OTHERS
    }
]

Dataset.add_dataset_columns(dataset_name="my_dataset", columns=columns_dict)

# Add rows using class method
rows_dict = [
    {
        "cells": [
            {
                "column_name": "text",
                "value": "Hello, world!"
            },
            {
                "column_name": "label",
                "value": 1
            }
        ]
    }
]

Dataset.add_dataset_rows(dataset_name="my_dataset", rows=rows_dict)
```

Here's a complete example demonstrating the full workflow:

<Tab title="Python">
```python
import os
from fi.datasets import Dataset
from fi.datasets.types import (
    DatasetConfig, Column, Row, Cell,
    DataTypeChoices, SourceChoices, ModelTypes
)

# Set up environment
os.environ["FI_API_KEY"] = "your_api_key"
os.environ["FI_SECRET_KEY"] = "your_secret_key"

def create_dataset_with_sdk():
    try:
        # 1. Create dataset configuration
        config = DatasetConfig(
            name="customer_feedback_dataset",
            model_type=ModelTypes.GENERATIVE_LLM
        )
        
        # 2. Initialize dataset
        dataset = Dataset(dataset_config=config)
        dataset = dataset.create()
        print(f"✓ Dataset created: {dataset.dataset_config.name}")
        
        # 3. Define columns
        columns = [
            Column(
                name="customer_query",
                data_type=DataTypeChoices.TEXT,
                source=SourceChoices.OTHERS
            ),
            Column(
                name="ai_response",
                data_type=DataTypeChoices.TEXT,
                source=SourceChoices.OTHERS
            ),
            Column(
                name="satisfaction_score",
                data_type=DataTypeChoices.INTEGER,
                source=SourceChoices.OTHERS
            ),
            Column(
                name="response_time_ms",
                data_type=DataTypeChoices.FLOAT,
                source=SourceChoices.OTHERS
            )
        ]
        
        # 4. Add columns
        dataset = dataset.add_columns(columns=columns)
        print("✓ Columns added successfully")
        
        # 5. Add sample data
        rows = [
            Row(
                order=1,
                cells=[
                    Cell(column_name="customer_query", value="How do I reset my password?"),
                    Cell(column_name="ai_response", value="You can reset your password by clicking the 'Forgot Password' link on the login page."),
                    Cell(column_name="satisfaction_score", value=8),
                    Cell(column_name="response_time_ms", value=1200.5)
                ]
            ),
            Row(
                order=2,
                cells=[
                    Cell(column_name="customer_query", value="What are your business hours?"),
                    Cell(column_name="ai_response", value="Our business hours are Monday to Friday, 9 AM to 6 PM EST."),
                    Cell(column_name="satisfaction_score", value=9),
                    Cell(column_name="response_time_ms", value=800.2)
                ]
            )
        ]
        
        dataset = dataset.add_rows(rows=rows)
        print("✓ Rows added successfully")
        
        # 6. Download and verify
        df = dataset.download(load_to_pandas=True)
        print(f"✓ Dataset shape: {df.shape}")
        print("✓ Dataset created and populated successfully!")
        
        return dataset
        
    except Exception as e:
        print(f"✗ Error creating dataset: {e}")
        return None

# Run the example
if __name__ == "__main__":
    dataset = create_dataset_with_sdk()
```

// Note: TypeScript SDK example for direct API calls
const addColumnsRequest = async (datasetId: string) => {
    const url = `https://api.futureagi.com/model-hub/develops/${datasetId}/add_columns/`;

    const requestBody = {
        new_columns_data: [
            {
                name: "customer_query",
                data_type: "text",
                source: "others"
            },
            {
                name: "satisfaction_score",
                data_type: "integer",
                source: "others"
            }
        ]
    };

    const response = await fetch(url, {
        method: "POST",
        headers: {
            "X-Api-Key": "your_api_key",
            "X-Secret-Key": "your_secret_key",
            "Content-Type": "application/json"
        },
        body: JSON.stringify(requestBody)
    });

    return await response.json();
};

const addRowsRequest = async (datasetId: string) => {
    const url = `https://api.futureagi.com/model-hub/develops/${datasetId}/add_rows/`;

    const requestBody = {

                cells: [

                        value: "How do I reset my password?"

                        column_name: "satisfaction_score",
                        value: 8
                    }

            "Content-Type": "application/json"

    });

    return await response.json();
};

## **6. Error Handling and Best Practices**

### **Error Handling**

```python
from fi.utils.errors import DatasetError, DatasetValidationError

try:
    dataset = Dataset(dataset_config=config)
    dataset = dataset.create()
    
    # Add columns and rows
    dataset = dataset.add_columns(columns=columns)
    dataset = dataset.add_rows(rows=rows)
    
except DatasetValidationError as e:
    print(f"Validation error: {e}")
except DatasetError as e:
    print(f"Dataset error: {e}")
except Exception as e:
    print(f"Unexpected error: {e}")
```

### **Best Practices**

- **Use Environment Variables**: Store API credentials securely
- **Validate Data**: Ensure data types match column definitions
- **Handle Errors**: Implement proper error handling for robust applications
- **Use Type Safety**: Leverage the SDK's type system for better code quality
- **Test with Small Datasets**: Start with small datasets to validate your structure
